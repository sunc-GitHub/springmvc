1. 在springmvc配置文件中

```java
<!--加载资源文件ReloadableResourceBundleMessageSource-->
<bean id="messageSource" class="org.springframework.context.support.ReloadableResourceBundleMessageSource">
    <property name="basenames">
        <list>
         <!--资源路径-->
       		<value>/WEB-INF/resource/message</value>
        </list>
    </property>
</bean>

<!--配置locale(一般用sessionLocate即在session存locale值)-->
<bean id="localeResolver" class="org.springframework.web.servlet.i18n.SessionLocaleResolver">
	<!--配置默认的locale-->
    <property name="defaultLocale" value="zh_CN"/>
	<!--配置session中参数名-->
    <property name="localeAttributeName" value="language" />
</bean>
		
<!--配置locale的拦截器，前端通常会?拼一个参数，这个参数就是locale字符串(zh_CN)-->
<mvc:interceptors>
    <bean class="org.springframework.web.servlet.i18n.LocaleChangeInterceptor">
		<!--前端传过来的参数要在这里配置,这里是url?currentLocale=.....-->
    	<property name="paramName" value="currentLocale"></property>
    </bean>
</mvc:interceptors>
```



2. 资源文件的创建
   	idea下直接File->new->Resource Bundle->添加自己的locale->设置资源名称

3. 前端jsp显示
   	<spring:资源名称 code="自定义国际化字符串" />
